ACCOUNT=$(aws sts get-caller-identity --query Account --output text)
REG=$ACCOUNT.dkr.ecr.us-east-1.amazonaws.com
kubectl create namespace tienda 2>/dev/null || true
cat <<EOF | kubectl apply -f -
apiVersion: apps/v1
kind: Deployment
metadata: { name: tienda-db, namespace: tienda }
spec:
  replicas: 1
  selector: { matchLabels: { app: tienda-db } }
  template:
    metadata: { labels: { app: tienda-db } }
    spec:
      containers:
        - name: db
          image: $REG/tienda-tech-db:latest
          ports: [ { containerPort: 3306 } ]
---
apiVersion: v1
kind: Service
metadata: { name: tienda-db, namespace: tienda }
spec:
  selector: { app: tienda-db }
  ports: [ { port: 3306, targetPort: 3306 } ]
---
apiVersion: apps/v1
kind: Deployment
metadata: { name: tienda-backend, namespace: tienda }
spec:
  replicas: 1
  selector: { matchLabels: { app: tienda-backend } }
  template:
    metadata: { labels: { app: tienda-backend } }
    spec:
      containers:
        - name: backend
          image: $REG/tienda-tech-backend:latest
          ports: [ { containerPort: 3001 } ]
          env:
            - { name: DB_HOST, value: tienda-db }
            - { name: DB_USER, value: alumno }
            - { name: DB_PASSWORD, value: alumno123 }
            - { name: DB_NAME, value: tienda_tecnologica }
            - { name: DB_PORT, value: "3306" }
---
apiVersion: v1
kind: Service
metadata: { name: tienda-backend, namespace: tienda }
spec:
  type: LoadBalancer
  selector: { app: tienda-backend }
  ports: [ { port: 3001, targetPort: 3001 } ]
---
apiVersion: apps/v1
kind: Deployment
metadata: { name: tienda-frontend, namespace: tienda }
spec:
  replicas: 1
  selector: { matchLabels: { app: tienda-frontend } }
  template:
    metadata: { labels: { app: tienda-frontend } }
    spec:
      containers:
        - name: frontend
          image: $REG/tienda-tech-frontend:latest
          ports: [ { containerPort: 80 } ]
---
apiVersion: v1
kind: Service
metadata: { name: tienda-frontend, namespace: tienda }
spec:
  type: LoadBalancer
  selector: { app: tienda-frontend }
  ports: [ { port: 80, targetPort: 80 } ]
EOF

4. FIX DE CODIGO SE DEJA EN UN MISMO LOAD BALANCER EL PUERTO DEL BACK CON EL DEL FRONT

ACCOUNT=$(aws sts get-caller-identity --query Account --output text)
REG=$ACCOUNT.dkr.ecr.us-east-1.amazonaws.com
kubectl delete deploy tienda-frontend tienda-backend -n tienda
kubectl delete svc  tienda-frontend tienda-backend -n tienda
cat <<EOF | kubectl apply -f -
apiVersion: apps/v1
kind: Deployment
metadata: { name: tienda-app, namespace: tienda }
spec:
  replicas: 1
  selector: { matchLabels: { app: tienda-app } }
  template:
    metadata: { labels: { app: tienda-app } }
    spec:
      containers:
        - name: backend
          image: $REG/tienda-tech-backend:latest
          ports: [ { containerPort: 3001 } ]
          env:
            - { name: DB_HOST, value: tienda-db }
            - { name: DB_USER, value: alumno }
            - { name: DB_PASSWORD, value: alumno123 }
            - { name: DB_NAME, value: tienda_tecnologica }
            - { name: DB_PORT, value: "3306" }
        - name: frontend
          image: $REG/tienda-tech-frontend:latest
          ports: [ { containerPort: 80 } ]
---
apiVersion: v1
kind: Service
metadata: { name: tienda-app, namespace: tienda }
spec:
  type: LoadBalancer
  selector: { app: tienda-app }
  ports:
    - { name: web, port: 80,   targetPort: 80 }
    - { name: api, port: 3001, targetPort: 3001 }
EOF

kubectl get svc tienda-app -n tienda
